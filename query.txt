select PARTITIONS.PART_ID, SDS.SD_ID, SDS.CD_ID, SERDES.SERDE_ID, PARTITIONS.CREATE_TIME, PARTITIONS.LAST_ACCESS_TIME, SDS.INPUT_FORMAT, SDS.IS_COMPRESSED, SDS.IS_STOREDASSUBDIRECTORIES, SDS.LOCATION, SDS.NUM_BUCKETS, SDS.OUTPUT_FORMAT, SERDES.NAME, SERDES.SLIB from PARTITIONS  left outer join SDS on PARTITIONS.SD_ID = SDS.SD_ID   left outer join SERDES on SDS.SERDE_ID = SERDES.SERDE_ID where PART_ID in (12714518) order by PART_NAME asc;
select SD_ID, SKEWED_COL_NAME from SKEWED_COL_NAMES where SD_ID in (13956042) and INTEGER_IDX >= 0 order by SD_ID asc, INTEGER_IDX asc;
select SERDE_ID, PARAM_KEY, PARAM_VALUE from SERDE_PARAMS where SERDE_ID in (13772071) and PARAM_KEY is not null order by SERDE_ID asc;
